│  │              ├─ 2.file         # Träger der langen Programm-Version zu "Kapitel 2"
│  │              ├─ 3.file         # Träger der langen Programm-Version zu "Kapitel 3"
│  │              ├─ 4.file         # Träger der langen Programm-Version zu "Kapitel 4"
│  │              ├─ Q.file         # Träger der langen Programm-Version zu "Kapitel Q" (Q&A-Kapitel, um direkt vor Ort mögliche Unklarheiten zu klären)
│  │              └─ z.file         # Träger der langen Programm-Version zu "Kapitel z" (Abschluss-Kapitel)
